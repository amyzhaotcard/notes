# 获取数组中最大的 2 个数，要求复杂难度最低

# 方法一：

var arr = [1, 5, 3, 2, 10, 15];
function maxarr(arr, len) {
  var max = [];
  arr.sort(function (a, b) {
    return a - b;
  }); // [1, 2, 3, 5, 10, 15]
  for (var i = 0; i < len; i++) {
    max.push(arr.pop());
  } // [15, 10]
  return max.reverse();
}

console.log(maxarr(arr, 2));

# 方法二：

var arr = [1, 5, 3, 2, 10, 15];
function maxarr(arr, len) {
  arr.sort(function (a, b) {
    return b - a;
  }); // [15, 10, 5, 3, 2, 1]
  return arr.slice(0, len);
}

console.log(maxarr(arr, 2));
